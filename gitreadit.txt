git 常用命令
------------------------------------------------------------------------
git config --global user.name ***
git config --global user.email ***
git init //初始化仓库
git add (目录/文件/.)
git commit -m "日志"


git remote add origin https://xxxxxxxxxxxx.git//关联远程代码仓库

git pull origin master --allow-unrelated-histories//强制合并两个不同版本库
